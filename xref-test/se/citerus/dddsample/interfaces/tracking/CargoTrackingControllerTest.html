44          CargoTrackingController controller = new CargoTrackingController(cargoRepository,
45              new HandlingEventRepositoryInMem(),

47              new TrackCommandValidator());

49          InternalResourceViewResolver resolver = new InternalResourceViewResolver();
50          resolver.setPrefix("/jsp/");
51          resolver.setSuffix(".jsp");
52          this.mockMvc = MockMvcBuilders.standaloneSetup(controller).setViewResolvers(resolver).build();
53      }

55      @Test
56      public void canGetCargo() throws Exception {
57          String trackingId = "ABC";
58          Map<String, Object> model = mockMvc.perform(post("/track").param("trackingId", trackingId)).andReturn().getModelAndView().getModel();
59          CargoTrackingViewAdapter cargoTrackingViewAdapter = (CargoTrackingViewAdapter) model.get("cargo");
60          assertThat(cargoTrackingViewAdapter.getTrackingId()).isEqualTo(trackingId);
61      }
62  
63      @Test
64      public void cannotGetUnknownCargo() throws Exception {
65          String trackingId = "UNKNOWN";
66          Map<String, Object> model = mockMvc.perform(post("/track").param("trackingId", trackingId)).andReturn().getModelAndView().getModel();
67          Errors errors = (Errors) model.get(BindingResult.MODEL_KEY_PREFIX + "trackCommand");
68          FieldError fe = errors.getFieldError("trackingId");
69          assertThat(fe.getCode()).isEqualTo("cargo.unknown_id");
70          assertThat(fe.getArguments().length).isEqualTo(1);
71          assertThat(fe.getArguments()[0]).isEqualTo(trackingId);
72      }
