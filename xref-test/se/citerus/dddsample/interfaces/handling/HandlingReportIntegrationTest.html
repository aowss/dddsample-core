34  @ExtendWith(SpringExtension.class)

36  @DirtiesContext(classMode = DirtiesContext.ClassMode.BEFORE_EACH_TEST_METHOD)
37  public class HandlingReportIntegrationTest {
38  
39      @LocalServerPort
40      private int port;
41  
42      @Autowired
43      private HandlingEventRepository repo;
44  
45      private final RestTemplate restTemplate = new RestTemplate();
46      private final ObjectMapper mapper = new ObjectMapper();
47  
48      @Disabled //TODO investigate failure when not run in isolation
49      @Transactional
50      @Test
51      void shouldReturn201ResponseWhenHandlingReportIsSubmitted() throws Exception {
52          String body = mapper.writeValueAsString(ImmutableMap.of(
53                  "completionTime", "2022-10-30T13:37:00",
54                  "trackingIds", Collections.singletonList("ABC123"),
55                  "type", HandlingEvent.Type.CUSTOMS.name(),
56                  "unLocode", SampleLocations.DALLAS.unLocode()
57          ));
58          URI uri = new UriTemplate("http://localhost:{port}/dddsample/handlingReport").expand(port);
59          RequestEntity<String> request = RequestEntity
60                  .post(uri)
61                  .contentType(MediaType.APPLICATION_JSON)
62                  .body(body);
63  
64          ResponseEntity<String> response = restTemplate.exchange(request, String.class);
65          assertThat(response.getStatusCode().value()).isEqualTo(201);
66  
67          Thread.sleep(1000); // TODO replace with Awaitility
68  
69          HandlingHistory handlingHistory = repo.lookupHandlingHistoryOfCargo(new TrackingId("ABC123"));
70          HandlingEvent handlingEvent = handlingHistory.mostRecentlyCompletedEvent();
71          assertThat(handlingEvent.cargo().trackingId().idString()).isEqualTo("ABC123");
72          assertThat(handlingEvent)
73                  .extracting("type", "location.unlocode")
74                  .containsExactly(HandlingEvent.Type.CUSTOMS, "USDAL");
75      }
76  
77      @SuppressWarnings("unchecked")
78      @Test
79      void shouldReturnValidationErrorResponseWhenInvalidHandlingReportIsSubmitted() throws Exception {
80          String body = mapper.writeValueAsString(ImmutableMap.of(
81                  "completionTime", "invalid date",
82                  "trackingIds", Collections.singletonList("ABC123"),
83                  "type", HandlingEvent.Type.CUSTOMS.name(),
84                  "unLocode", SampleLocations.STOCKHOLM.code(),
85                  "voyageNumber", "0101"
86          ));
87  
88          URI uri = new UriTemplate("http://localhost:{port}/dddsample/handlingReport").expand(port);
89          RequestEntity<String> request = RequestEntity
90                  .post(uri)
91                  .contentType(MediaType.APPLICATION_JSON)
92                  .body(body);
93          try {
94              restTemplate.exchange(request, String.class);
95              fail("Did not throw HttpClientErrorException");
96          } catch (HttpClientErrorException e) {
97              Map<String, String> map = mapper.readValue(e.getResponseBodyAsString(), Map.class);
98              assertThat(map.get("message")).contains("JSON parse error: Cannot deserialize value of type `java.time.LocalDateTime` from String \"invalid date\"");
99              assertThat(map.get("message")).contains("Text 'invalid date' could not be parsed at index 0");
100         }
101     }
