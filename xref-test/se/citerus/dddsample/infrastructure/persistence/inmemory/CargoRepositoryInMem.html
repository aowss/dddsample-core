16  /**
17   * CargoRepositoryInMem implement the CargoRepository interface but is a test
18   * class not intended for usage in real application.
19   * <p>
20   * It setup a simple local hash with a number of Cargo's with TrackingId as key
21   * defined at compile time.
22   * <p>
23   */
24  public class CargoRepositoryInMem implements CargoRepository {
25  
26      private final Map<String, Cargo> cargoDb;
27      private HandlingEventRepository handlingEventRepository;
28  
29      /**
30       * Constructor.
31       */
32      public CargoRepositoryInMem() {
33          cargoDb = new HashMap<>();
34      }
35  
36      public Cargo find(final TrackingId trackingId) {
37          return cargoDb.get(trackingId.idString());
38      }
39  
40      public void store(final Cargo cargo) {
41          cargoDb.put(cargo.trackingId().idString(), cargo);
42      }
43  
44      public TrackingId nextTrackingId() {
45          String random = UUID.randomUUID().toString().toUpperCase();
46          return new TrackingId(
47                  random.substring(0, random.indexOf("-"))
48          );
49      }
50  
51      public List<Cargo> getAll() {
52          return new ArrayList<>(cargoDb.values());
53      }
54  
55      public void init() throws Exception {
56          final TrackingId xyz = new TrackingId("XYZ");
57          final Cargo cargoXYZ = createCargoWithDeliveryHistory(
58                  xyz, STOCKHOLM, MELBOURNE, handlingEventRepository.lookupHandlingHistoryOfCargo(xyz));
59          cargoDb.put(xyz.idString(), cargoXYZ);
60  
61          final TrackingId zyx = new TrackingId("ZYX");
62          final Cargo cargoZYX = createCargoWithDeliveryHistory(
63                  zyx, MELBOURNE, STOCKHOLM, handlingEventRepository.lookupHandlingHistoryOfCargo(zyx));
64          cargoDb.put(zyx.idString(), cargoZYX);
65  
66          final TrackingId abc = new TrackingId("ABC");
67          final Cargo cargoABC = createCargoWithDeliveryHistory(
68                  abc, STOCKHOLM, HELSINKI, handlingEventRepository.lookupHandlingHistoryOfCargo(abc));
69          cargoDb.put(abc.idString(), cargoABC);
70  
71          final TrackingId cba = new TrackingId("CBA");
72          final Cargo cargoCBA = createCargoWithDeliveryHistory(
73                  cba, HELSINKI, STOCKHOLM, handlingEventRepository.lookupHandlingHistoryOfCargo(cba));
74          cargoDb.put(cba.idString(), cargoCBA);
75      }
76  
77      public void setHandlingEventRepository(final HandlingEventRepository handlingEventRepository) {
78          this.handlingEventRepository = handlingEventRepository;
79      }
80  
81      public static Cargo createCargoWithDeliveryHistory(TrackingId trackingId,
82                                                         Location origin,
83                                                         Location destination,
84                                                         HandlingHistory handlingHistory) {
85  
86          final RouteSpecification routeSpecification = new RouteSpecification(origin, destination, Instant.now());
87          final Cargo cargo = new Cargo(trackingId, routeSpecification);
88          cargo.deriveDeliveryProgress(handlingHistory);
89  
90          return cargo;
91      }
92  }
