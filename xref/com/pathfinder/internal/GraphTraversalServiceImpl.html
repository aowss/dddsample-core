23    public List<TransitPath> findShortestPath(
24        final String originNode, final String destinationNode, final Properties limitations) {
25      List<String> allVertices = dao.listAllNodes();
26      allVertices.remove(originNode);
27      allVertices.remove(destinationNode);
28  
29      int candidateCount = getRandomNumberOfCandidates();
30      List<TransitPath> candidates = new ArrayList<>(candidateCount);
31  
32      for (int i = 0; i < candidateCount; i++) {
33        allVertices = getRandomChunkOfNodes(allVertices);
34        List<TransitEdge> transitEdges = new ArrayList<>(allVertices.size() - 1);
35        String fromNode = originNode;
36        Instant date = Instant.now();
37  
38        for (int j = 0; j <= allVertices.size(); ++j) {
39          Instant fromDate = nextDate(date);
40          Instant toDate = nextDate(fromDate);
41          String toNode = (j >= allVertices.size() ? destinationNode : allVertices.get(j));
42          transitEdges.add(
43              new TransitEdge(
44                  dao.getTransitEdge(fromNode, toNode), fromNode, toNode, fromDate, toDate));
45          fromNode = toNode;
46          date = nextDate(toDate);
47        }
48        candidates.add(new TransitPath(transitEdges));
49      }
50      return candidates;
51    }
52  
53    private Instant nextDate(Instant date) {
54      return date.plus(1, ChronoUnit.DAYS).plus((random.nextInt(1000) - 500), ChronoUnit.MINUTES);
55    }
56  
57    private int getRandomNumberOfCandidates() {
58      return 3 + random.nextInt(3);
59    }
60  
61    private List<String> getRandomChunkOfNodes(List<String> allNodes) {
62      Collections.shuffle(allNodes);
63      final int total = allNodes.size();
64      final int chunk = total > 4 ? 1 + random.nextInt(5) : total;
65      return allNodes.subList(0, chunk);
66    }
