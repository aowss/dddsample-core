4   import org.apache.commons.lang3.Validate;
5   import org.apache.commons.lang3.builder.EqualsBuilder;
6   import org.apache.commons.lang3.builder.HashCodeBuilder;
7   import se.citerus.dddsample.domain.model.handling.HandlingEvent;
8   import se.citerus.dddsample.domain.model.location.Location;
9   import se.citerus.dddsample.domain.model.voyage.Voyage;

12  import java.util.Objects;
13  
14  /**
15   * A handling activity represents how and where a cargo can be handled,
16   * and can be used to express predictions about what is expected to
17   * happen to a cargo in the future.
18   *
19   */
20  @Embeddable
21  public class HandlingActivity implements ValueObject<HandlingActivity> {
22  
23    // TODO make HandlingActivity a part of HandlingEvent too? There is some overlap. 
24  
25    @Enumerated(value = EnumType.STRING)
26    @Column(name = "next_expected_handling_event_type")
27    public HandlingEvent.Type type;
28  
29    @ManyToOne()
30    @JoinColumn(name = "next_expected_location_id")
31    public Location location;
32  
33    @ManyToOne
34    @JoinColumn(name = "next_expected_voyage_id")
35    public Voyage voyage;
36  
37    public HandlingActivity(final HandlingEvent.Type type, final Location location) {
38      Objects.requireNonNull(type, "Handling event type is required");
39      Objects.requireNonNull(location, "Location is required");
40  
41      this.type = type;
42      this.location = location;
43    }
44  
45    public HandlingActivity(final HandlingEvent.Type type, final Location location, final Voyage voyage) {
46      Objects.requireNonNull(type, "Handling event type is required");
47      Objects.requireNonNull(location, "Location is required");
48      Objects.requireNonNull(location, "Voyage is required");
49  
50      this.type = type;
51      this.location = location;
52      this.voyage = voyage;
53    }
54  
55    public HandlingEvent.Type type() {
56      return type;
57    }
58  
59    public Location location() {
60      return location;
61    }
62  
63    public Voyage voyage() {
64      return voyage;
65    }
66  
67    @Override
68    public boolean sameValueAs(final HandlingActivity other) {
69      return other != null && new EqualsBuilder().
70        append(this.type, other.type).
71        append(this.location, other.location).
72        append(this.voyage, other.voyage).
73        isEquals();
74    }
75  
76    @Override
77    public int hashCode() {
78      return new HashCodeBuilder().
79        append(this.type).
80        append(this.location).
81        append(this.voyage).
82        toHashCode();
83    }
84  
85    @Override
86    public boolean equals(final Object obj) {
87      if (obj == this) return true;
88      if (obj == null) return false;
89      if (obj.getClass() != this.getClass()) return false;
90  
91      HandlingActivity other = (HandlingActivity) obj;
92  
93      return sameValueAs(other);
94    }
95  
96    protected HandlingActivity() {
97      // Needed by Hibernate
98    }
99  
100 }
