3   import jakarta.persistence.*;
4   import org.apache.commons.lang3.Validate;
5   import se.citerus.dddsample.domain.shared.DomainEntity;

7   import java.util.Objects;

9   /**
10   * A location is our model is stops on a journey, such as cargo
11   * origin or destination, or carrier movement endpoints.
12   * It is uniquely identified by a UN Locode.
13   *
14   */
15  @Entity(name = "Location")
16  @Table(name = "Location")
17  public final class Location implements DomainEntity<Location> {
18  
19    @Id
20    @GeneratedValue(strategy = GenerationType.AUTO)
21    private long id;
22  
23    @Column(nullable = false, unique = true, updatable = false)
24    private String unlocode;
25  
26    @Column(nullable = false)
27    private String name;
28  
29    /**
30     * Special Location object that marks an unknown location.
31     */
32    public static final Location UNKNOWN = new Location(
33      new UnLocode("XXXXX"), "Unknown location"
34    );
35  
36    /**
37     * Package-level constructor, visible for test and sample data purposes.
38     *
39     * @param unLocode UN Locode
40     * @param name     location name
41     * @throws IllegalArgumentException if the UN Locode or name is null
42     */
43    public Location(final UnLocode unLocode, final String name) {
44      Objects.requireNonNull(unLocode);
45      Objects.requireNonNull(name);
46      
47      this.unlocode = unLocode.idString();
48      this.name = name;
49    }
50  
51    // Used by JPA
52    public Location(String unloCode, String name) {
53      this.unlocode = unloCode;
54      this.name = name;
55    }
56  
57    /**
58     * @return UN Locode for this location.
59     */
60    public UnLocode unLocode() {
61      return new UnLocode(unlocode);
62    }
63  
64    /**
65     * @return Actual name of this location, e.g. "Stockholm".
66     */
67    public String name() {
68      return name;
69    }
70  
71    public String code() {
72      return unlocode;
73    }
74  
75    public long id() {
76      return id;
77    }
78  
79    /**
80     * @param object to compare
81     * @return Since this is an entiy this will be true iff UN locodes are equal.
82     */
83    @Override
84    public boolean equals(final Object object) {
85      if (object == null) {
86        return false;
87      }
88      if (this == object) {
89        return true;
90      }
91      if (!(object instanceof Location other)) {
92        return false;
93      }
94        return sameIdentityAs(other);
95    }
96  
97    @Override
98    public boolean sameIdentityAs(final Location other) {
99      return this.unlocode.equals(other.unlocode);
100   }
101 
102   /**
103    * @return Hash code of UN locode.
104    */
105   @Override
106   public int hashCode() {
107     return unlocode.hashCode();
108   }
109 
110   @Override
111   public String toString() {
112     return name + " [" + unlocode + "]";
113   }
114 
115   Location() {
116     // Needed by Hibernate
117   }
118 
119 }
