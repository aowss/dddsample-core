4   import org.slf4j.LoggerFactory;
5   import org.springframework.transaction.annotation.Transactional;
6   import se.citerus.dddsample.application.BookingService;
7   import se.citerus.dddsample.domain.model.cargo.*;
8   import se.citerus.dddsample.domain.model.location.Location;

10  import se.citerus.dddsample.domain.model.location.UnLocode;
11  import se.citerus.dddsample.domain.service.RoutingService;
12  
13  import java.lang.invoke.MethodHandles;
14  import java.time.Instant;
15  import java.util.Collections;
16  import java.util.List;
17  
18  public class BookingServiceImpl implements BookingService {
19  
20    private final CargoRepository cargoRepository;
21    private final LocationRepository locationRepository;
22    private final RoutingService routingService;
23    private final CargoFactory cargoFactory;
24    private static final Logger logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
25  
26    public BookingServiceImpl(final CargoRepository cargoRepository,
27                              final LocationRepository locationRepository,
28                              final RoutingService routingService,
29                              final CargoFactory cargoFactory) {
30      this.cargoRepository = cargoRepository;
31      this.locationRepository = locationRepository;
32      this.routingService = routingService;
33      this.cargoFactory = cargoFactory;
34    }
35  
36    @Override
37    @Transactional
38    public TrackingId bookNewCargo(final UnLocode originUnLocode,
39                                   final UnLocode destinationUnLocode,
40                                   final Instant arrivalDeadline) {
41      Cargo cargo = cargoFactory.createCargo(originUnLocode, destinationUnLocode, arrivalDeadline);
42  
43      cargoRepository.store(cargo);
44      logger.info("Booked new cargo with tracking id {}", cargo.trackingId().idString());
45  
46      return cargo.trackingId();
47    }
48  
49    @Override
50    @Transactional
51    public List<Itinerary> requestPossibleRoutesForCargo(final TrackingId trackingId) {
52      final Cargo cargo = cargoRepository.find(trackingId);
53  
54      if (cargo == null) {
55        return Collections.emptyList();
56      }
57  
58      return routingService.fetchRoutesForSpecification(cargo.routeSpecification());
59    }
60  
61    @Override
62    @Transactional
63    public void assignCargoToRoute(final Itinerary itinerary, final TrackingId trackingId) {
64      final Cargo cargo = cargoRepository.find(trackingId);
65      if (cargo == null) {
66        throw new IllegalArgumentException("Can't assign itinerary to non-existing cargo " + trackingId);
67      }
68  
69      cargo.assignToRoute(itinerary);
70      cargoRepository.store(cargo);
71  
72      logger.info("Assigned cargo {} to new route", trackingId);
73    }
74  
75    @Override
76    @Transactional
77    public void changeDestination(final TrackingId trackingId, final UnLocode unLocode) {
78      final Cargo cargo = cargoRepository.find(trackingId);
79      final Location newDestination = locationRepository.find(unLocode);
80  
81      final RouteSpecification routeSpecification = new RouteSpecification(
82        cargo.origin(), newDestination, cargo.routeSpecification().arrivalDeadline()
83      );
84      cargo.specifyNewRoute(routeSpecification);
85  
86      cargoRepository.store(cargo);
87      logger.info("Changed destination for cargo {} to {}", trackingId, routeSpecification.destination());
88    }
89  
90  }
