30                                              @NonNull Instant completionTime,
31                                              @NonNull TrackingId trackingId,
32                                              @NonNull VoyageNumber voyageNumber,
33                                              @NonNull HandlingEvent.Type type,
34                                              @NonNull UnLocode unLocode) {
