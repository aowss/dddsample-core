21  /**
22   * Controller for tracking cargo. This interface sits immediately on top of the
23   * domain layer, unlike the booking interface which has a a remote facade and supporting
24   * DTOs in between.
25   * <p>
26   * An adapter class, designed for the tracking use case, is used to wrap the domain model
27   * to make it easier to work with in a web page rendering context. We do not want to apply
28   * view rendering constraints to the design of our domain model, and the adapter
29   * helps us shield the domain model classes.
30   * <p>
31   *
32   * @see CargoTrackingViewAdapter
33   * @see se.citerus.dddsample.interfaces.booking.web.CargoAdminController
34   */
35  @Controller
36  @RequestMapping("/track")
37  public final class CargoTrackingController {
38  
39      private final CargoRepository cargoRepository;
40      private final HandlingEventRepository handlingEventRepository;
41      private final MessageSource messageSource;
42      private final TrackCommandValidator trackCommandValidator;
43  
44      public CargoTrackingController(@NonNull CargoRepository cargoRepository,
45                                     @NonNull HandlingEventRepository handlingEventRepository,
46                                     @NonNull MessageSource messageSource,
47                                     @NonNull TrackCommandValidator trackCommandValidator) {
48          this.cargoRepository = cargoRepository;
49          this.handlingEventRepository = handlingEventRepository;
50          this.messageSource = messageSource;
51          this.trackCommandValidator = trackCommandValidator;
52      }
53  
54      @RequestMapping(method = RequestMethod.GET)
55      public String index(final Map<String, Object> model) {
56          // using the empty command to support the thymeleaf form `<form method="post" th:object="${trackCommand}">`
57          model.put("trackCommand", new TrackCommand());
58          return "track";
59      }
60  
61      @RequestMapping(method = RequestMethod.POST)
62      private String onSubmit(final HttpServletRequest request,
63                              final TrackCommand command,
64                              final Map<String, Object> model,
65                              final BindingResult bindingResult) {
66          trackCommandValidator.validate(command, bindingResult);
67  
68          final TrackingId trackingId = new TrackingId(command.getTrackingId());
69          final Cargo cargo = cargoRepository.find(trackingId);
70  
71          if (cargo != null) {
72              final Locale locale = RequestContextUtils.getLocale(request);
73              final List<HandlingEvent> handlingEvents = handlingEventRepository.lookupHandlingHistoryOfCargo(trackingId).distinctEventsByCompletionTime();
74              model.put("cargo", new CargoTrackingViewAdapter(cargo, messageSource, locale, handlingEvents));
75          } else {
76              bindingResult.rejectValue("trackingId", "cargo.unknown_id", new Object[]{command.getTrackingId()}, "Unknown tracking id");
77          }
78          return "track";
79      }
80  }
