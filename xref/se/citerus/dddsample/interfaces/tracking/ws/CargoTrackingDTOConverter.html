17  public class CargoTrackingDTOConverter {
18      private static final DateTimeFormatter formatter = DateTimeFormatter
19              .ofPattern("MMM d, uuuu, h:mm:ss a", Locale.ENGLISH) // See https://github.com/spring-projects/spring-framework/wiki/Date-and-Time-Formatting-with-JDK-20-and-higher#recommendations
20              .withZone(ZoneOffset.UTC);
21  
22      public static CargoTrackingDTO convert(Cargo cargo, List<HandlingEvent> handlingEvents, MessageSource messageSource, Locale locale) {
23          List<HandlingEventDTO> handlingEventDTOs = convertHandlingEvents(handlingEvents, cargo, messageSource, locale);
24          return new CargoTrackingDTO(
25                  convertTrackingId(cargo),
26                  convertStatusText(cargo, messageSource, locale),
27                  convertDestination(cargo),
28                  convertEta(cargo),
29                  convertNextExpectedActivity(cargo),
30                  convertIsMisdirected(cargo),
31                  handlingEventDTOs);
32      }
33  
34      private static List<HandlingEventDTO> convertHandlingEvents(List<HandlingEvent> handlingEvents, Cargo cargo, MessageSource messageSource, Locale locale) {
35          return handlingEvents.stream().map(he -> new HandlingEventDTO(
36                  convertLocation(he),
37                  convertTime(he),
38                  convertType(he),
39                  convertVoyageNumber(he),
40                  convertIsExpected(he, cargo),
41                  convertDescription(he, messageSource, locale)
42          )).collect(Collectors.toList());
43      }
44  
45      protected static String convertDescription(HandlingEvent handlingEvent, MessageSource messageSource, Locale locale) {
46          Object[] args;
47  
48          switch (handlingEvent.type()) {
49              case LOAD:
50              case UNLOAD:
51                  args = new Object[]{
52                          handlingEvent.voyage().voyageNumber().idString(),
53                          handlingEvent.location().name(),
54                          formatter.format(handlingEvent.completionTime())
55                  };
56                  break;
57              case RECEIVE:
58              case CUSTOMS:
59              case CLAIM:
60                  args = new Object[]{
61                          handlingEvent.location().name(),
62                          formatter.format(handlingEvent.completionTime())
63                  };
64                  break;
65  
66              default:
67                  args = new Object[]{};
68          }
69  
70          String key = "deliveryHistory.eventDescription." + handlingEvent.type().name();
71  
72          return messageSource.getMessage(key, args, locale);
73      }
74  
75      private static boolean convertIsExpected(HandlingEvent handlingEvent, Cargo cargo) {
76          return cargo.itinerary().isExpected(handlingEvent);
77      }
78  
79      private static String convertVoyageNumber(HandlingEvent handlingEvent) {
80          return handlingEvent.voyage().voyageNumber().idString();
81      }
82  
83      private static String convertType(HandlingEvent handlingEvent) {
84          return handlingEvent.type().toString();
85      }
86  
87      private static String convertTime(HandlingEvent handlingEvent) {
88          return formatter
89                  .format(handlingEvent.completionTime());
90      }
91  
92      private static String convertLocation(HandlingEvent handlingEvent) {
93          return handlingEvent.location().name();
94      }
95  
96      private static String convertTrackingId(Cargo cargo) {
97          return cargo.trackingId().idString();
98      }
99  
100     protected static String convertStatusText(Cargo cargo, MessageSource messageSource, Locale locale) {
101         final Delivery delivery = cargo.delivery();
102         final String code = "cargo.status." + delivery.transportStatus().name();
103 
104         final Object[] args;
105         switch (delivery.transportStatus()) {
106             case IN_PORT:
107                 args = new Object[]{delivery.lastKnownLocation().name()};
108                 break;
109             case ONBOARD_CARRIER:
110                 args = new Object[]{delivery.currentVoyage().voyageNumber().idString()};
111                 break;
112             case CLAIMED:
113             case NOT_RECEIVED:
114             case UNKNOWN:
115             default:
116                 args = null;
117                 break;
118         }
119 
120         return messageSource.getMessage(code, args, "[Unknown status]", locale);
121     }
122 
123     private static String convertDestination(Cargo cargo) {
124         return cargo.routeSpecification().destination().name();
125     }
126 
127     private static String convertEta(Cargo cargo) {
128         Instant date = cargo.delivery().estimatedTimeOfArrival();
129         return date == null ? "Unknown" : formatter.format(date);
130     }
131 
132     protected static String convertNextExpectedActivity(Cargo cargo) {
133         HandlingActivity activity = cargo.delivery().nextExpectedActivity();
134         if (activity == null) {
135             return "";
136         }
137 
138         String text = "Next expected activity is to ";
139         HandlingEvent.Type type = activity.type();
140         if (type.sameValueAs(HandlingEvent.Type.LOAD)) {
141             return
142                     text + type.name().toLowerCase() + " cargo onto voyage " + activity.voyage().voyageNumber() +
143                             " in " + activity.location().name();
144         } else if (type.sameValueAs(HandlingEvent.Type.UNLOAD)) {
145             return
146                     text + type.name().toLowerCase() + " cargo off of " + activity.voyage().voyageNumber() +
147                             " in " + activity.location().name();
148         } else {
149             return text + type.name().toLowerCase() + " cargo in " + activity.location().name();
150         }
151     }
152 
153     private static boolean convertIsMisdirected(Cargo cargo) {
154         return cargo.delivery().isMisdirected();
155     }
