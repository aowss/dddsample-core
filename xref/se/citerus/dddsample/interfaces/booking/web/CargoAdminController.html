26  /**
27   * Handles cargo booking and routing. Operates against a dedicated remoting service facade,
28   * and could easily be rewritten as a thick Swing client. Completely separated from the domain layer,
29   * unlike the tracking user interface.
30   * <p>
31   * In order to successfully keep the domain model shielded from user interface considerations,
32   * this approach is generally preferred to the one taken in the tracking controller. However,
33   * there is never any one perfect solution for all situations, so we've chosen to demonstrate
34   * two polarized ways to build user interfaces.
35   *
36   * @see se.citerus.dddsample.interfaces.tracking.CargoTrackingController
37   */
38  @Controller
39  @RequestMapping("/admin")
40  public final class CargoAdminController {
41  
42      private final BookingServiceFacade bookingServiceFacade;
43  
44      public CargoAdminController(BookingServiceFacade bookingServiceFacade) {
45          this.bookingServiceFacade = bookingServiceFacade;
46      }
47  
48      @InitBinder
49      private void initBinder(HttpServletRequest request, ServletRequestDataBinder binder) throws Exception {
50          binder.registerCustomEditor(Instant.class, new CustomDateEditor(new SimpleDateFormat("yyyy-MM-dd HH:mm"), false));
51      }
52  
53      @RequestMapping("/registration")
54      public String registration(HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) throws Exception {
55          List<LocationDTO> dtoList = bookingServiceFacade.listShippingLocations();
56  
57          List<String> unLocodeStrings = new ArrayList<String>();
58  
59          for (LocationDTO dto : dtoList) {
60              unLocodeStrings.add(dto.getUnLocode());
61          }
62  
63          model.put("unlocodes", unLocodeStrings);
64          model.put("locations", dtoList);
65          return "admin/registrationForm";
66      }
67  
68      @RequestMapping(value = "/register", method = RequestMethod.POST)
69      public void register(HttpServletRequest request, HttpServletResponse response,
70                           RegistrationCommand command) throws Exception {
71  
72          LocalDate arrivalDeadline = LocalDate.parse(command.getArrivalDeadline(), DateTimeFormatter.ofPattern("dd/MM/yyyy"));
73          String trackingId = bookingServiceFacade.bookNewCargo(
74              command.getOriginUnlocode(), command.getDestinationUnlocode(), arrivalDeadline.atStartOfDay().toInstant(ZoneOffset.UTC)
75          );
76          response.sendRedirect("show?trackingId=" + trackingId);
77      }
78  
79      @RequestMapping("/list")
80      public String list(HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) throws Exception {
81          List<CargoRoutingDTO> cargoList = bookingServiceFacade.listAllCargos();
82  
83          model.put("cargoList", cargoList);
84          return "admin/list";
85      }
86  
87      @RequestMapping("/show")
88      public String show(HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) throws Exception {
89          String trackingId = request.getParameter("trackingId");
90          CargoRoutingDTO dto = bookingServiceFacade.loadCargoForRouting(trackingId);
91          model.put("cargo", dto);
92          return "admin/show";
93      }
94  
95      @RequestMapping("/selectItinerary")
96      public String selectItinerary(HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) throws Exception {
97          String trackingId = request.getParameter("trackingId");
98  
99          List<RouteCandidateDTO> routeCandidates = bookingServiceFacade.requestPossibleRoutesForCargo(trackingId);
100         model.put("routeCandidates", routeCandidates);
101 
102         CargoRoutingDTO cargoDTO = bookingServiceFacade.loadCargoForRouting(trackingId);
103         model.put("cargo", cargoDTO);
104 
105         return "admin/selectItinerary";
106     }
107 
108     @RequestMapping(value = "/assignItinerary", method = RequestMethod.POST)
109     public void assignItinerary(HttpServletRequest request, HttpServletResponse response, RouteAssignmentCommand command) throws Exception {
110         List<LegDTO> legDTOs = new ArrayList<LegDTO>(command.getLegs().size());
111         for (RouteAssignmentCommand.LegCommand leg : command.getLegs()) {
112             legDTOs.add(new LegDTO(
113                 leg.getVoyageNumber(),
114                 leg.getFromUnLocode(),
115                 leg.getToUnLocode(),
116                 leg.getFromDate(),
117                 leg.getToDate())
118             );
119         }
120 
121         RouteCandidateDTO selectedRoute = new RouteCandidateDTO(legDTOs);
122 
123         bookingServiceFacade.assignCargoToRoute(command.getTrackingId(), selectedRoute);
124 
125         response.sendRedirect("show?trackingId=" + command.getTrackingId());
126     }
127 
128     @RequestMapping(value = "/pickNewDestination")
129     public String pickNewDestination(HttpServletRequest request, HttpServletResponse response, Map<String, Object> model) throws Exception {
130         List<LocationDTO> locations = bookingServiceFacade.listShippingLocations();
131         model.put("locations", locations);
132 
133         String trackingId = request.getParameter("trackingId");
134         CargoRoutingDTO cargo = bookingServiceFacade.loadCargoForRouting(trackingId);
135         model.put("cargo", cargo);
136 
137         return "admin/pickNewDestination";
138     }
139 
140     @RequestMapping(value = "/changeDestination", method = RequestMethod.POST)
141     public void changeDestination(HttpServletRequest request, HttpServletResponse response) throws Exception {
142         String trackingId = request.getParameter("trackingId");
143         String unLocode = request.getParameter("unlocode");
144         bookingServiceFacade.changeDestination(trackingId, unLocode);
145         response.sendRedirect("show?trackingId=" + trackingId);
146     }
147 }
