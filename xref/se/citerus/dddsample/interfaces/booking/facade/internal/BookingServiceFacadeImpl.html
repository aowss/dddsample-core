25  /**
26   * This implementation has additional support from the infrastructure, for exposing as an RMI
27   * service and for keeping the OR-mapper unit-of-work open during DTO assembly,
28   * analogous to the view rendering for web interfaces.
29   *
30   */
31  public class BookingServiceFacadeImpl implements BookingServiceFacade {
32  
33    private BookingService bookingService;
34    private LocationRepository locationRepository;
35    private CargoRepository cargoRepository;
36    private VoyageRepository voyageRepository;
37  
38    public BookingServiceFacadeImpl(BookingService bookingService, LocationRepository locationRepository, CargoRepository cargoRepository, VoyageRepository voyageRepository) {
39      this.bookingService = bookingService;
40      this.locationRepository = locationRepository;
41      this.cargoRepository = cargoRepository;
42      this.voyageRepository = voyageRepository;
43    }
44  
45    @Override
46    public List<LocationDTO> listShippingLocations() {
47      final List<Location> allLocations = locationRepository.getAll();
48      final LocationDTOAssembler assembler = new LocationDTOAssembler();
49      return assembler.toDTOList(allLocations);
50    }
51  
52    @Override
53    public String bookNewCargo(String origin, String destination, Instant arrivalDeadline) {
54      TrackingId trackingId = bookingService.bookNewCargo(
55        new UnLocode(origin), 
56        new UnLocode(destination),
57        arrivalDeadline
58      );
59      return trackingId.idString();
60    }
61  
62    @Override
63    public CargoRoutingDTO loadCargoForRouting(String trackingId) {
64      final Cargo cargo = cargoRepository.find(new TrackingId(trackingId));
65      final CargoRoutingDTOAssembler assembler = new CargoRoutingDTOAssembler();
66      return assembler.toDTO(cargo);
67    }
68  
69    @Override
70    public void assignCargoToRoute(String trackingIdStr, RouteCandidateDTO routeCandidateDTO) {
71      final Itinerary itinerary = new ItineraryCandidateDTOAssembler().fromDTO(routeCandidateDTO, voyageRepository, locationRepository);
72      final TrackingId trackingId = new TrackingId(trackingIdStr);
73  
74      bookingService.assignCargoToRoute(itinerary, trackingId);
75    }
76  
77    @Override
78    public void changeDestination(String trackingId, String destinationUnLocode) throws RemoteException {
79      bookingService.changeDestination(new TrackingId(trackingId), new UnLocode(destinationUnLocode));
80    }
81  
82    @Override
83    public List<CargoRoutingDTO> listAllCargos() {
84      final List<Cargo> cargoList = cargoRepository.getAll();
85      final List<CargoRoutingDTO> dtoList = new ArrayList<CargoRoutingDTO>(cargoList.size());
86      final CargoRoutingDTOAssembler assembler = new CargoRoutingDTOAssembler();
87      for (Cargo cargo : cargoList) {
88        dtoList.add(assembler.toDTO(cargo));
89      }
90      return dtoList;
91    }
92  
93    @Override
94    public List<RouteCandidateDTO> requestPossibleRoutesForCargo(String trackingId) throws RemoteException {
95      final List<Itinerary> itineraries = bookingService.requestPossibleRoutesForCargo(new TrackingId(trackingId));
96  
97      final List<RouteCandidateDTO> routeCandidates = new ArrayList<RouteCandidateDTO>(itineraries.size());
98      final ItineraryCandidateDTOAssembler dtoAssembler = new ItineraryCandidateDTOAssembler();
99      for (Itinerary itinerary : itineraries) {
100       routeCandidates.add(dtoAssembler.toDTO(itinerary));
101     }
102 
103     return routeCandidates;
104   }
105 }
