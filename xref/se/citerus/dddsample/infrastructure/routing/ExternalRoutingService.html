24  /**
25   * Our end of the routing service. This is basically a data model
26   * translation layer between our domain model and the API put forward
27   * by the routing team, which operates in a different context from us.
28   *
29   */
30  public class ExternalRoutingService implements RoutingService {
31    private static final Logger logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
32  
33    private final GraphTraversalService graphTraversalService;
34    private final LocationRepository locationRepository;
35    private final VoyageRepository voyageRepository;
36  
37    public ExternalRoutingService(GraphTraversalService graphTraversalService, LocationRepository locationRepository, VoyageRepository voyageRepository) {
38      this.graphTraversalService = graphTraversalService;
39      this.locationRepository = locationRepository;
40      this.voyageRepository = voyageRepository;
41    }
42  
43    public List<Itinerary> fetchRoutesForSpecification(RouteSpecification routeSpecification) {
44      /*
45        The RouteSpecification is picked apart and adapted to the external API.
46       */
47      final Location origin = routeSpecification.origin();
48      final Location destination = routeSpecification.destination();
49  
50      final Properties limitations = new Properties();
51      limitations.setProperty("DEADLINE", routeSpecification.arrivalDeadline().toString());
52  
53      final List<TransitPath> transitPaths;
54      transitPaths = graphTraversalService.findShortestPath(
55        origin.unLocode().idString(),
56        destination.unLocode().idString(),
57        limitations
58      );
59  
60      /*
61       The returned result is then translated back into our domain model.
62      */
63      final List<Itinerary> itineraries = transitPaths.stream()
64              .map(this::toItinerary)
65              .filter(itinerary -> isSatisfyingRouteSpec(itinerary, routeSpecification))
66              .collect(Collectors.toList());
67  
68      return itineraries;
69    }
70  
71    private static boolean isSatisfyingRouteSpec(Itinerary itinerary, RouteSpecification routeSpecification) {
72      if (routeSpecification.isSatisfiedBy(itinerary)) {
73        return true;
74      } else {
75        logger.warn("Received itinerary that did not satisfy the route specification");
76        return false;
77      }
78    }
79  
80    private Itinerary toItinerary(TransitPath transitPath) {
81      List<Leg> legs = new ArrayList<>(transitPath.getTransitEdges().size());
82      for (TransitEdge edge : transitPath.getTransitEdges()) {
83        legs.add(toLeg(edge));
84      }
85      return new Itinerary(legs);
86    }
87  
88    private Leg toLeg(TransitEdge edge) {
89      return new Leg(
90              voyageRepository.find(new VoyageNumber(edge.getEdge())),
91              locationRepository.find(new UnLocode(edge.getFromNode())),
92              locationRepository.find(new UnLocode(edge.getToNode())),
93              edge.getFromDate(),
94              edge.getToDate()
95      );
96    }
